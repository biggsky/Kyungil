# 포켓몬 뽑기 컨트랙트

# 리액트 설치

```sh
npx create-react-app erc20

리믹스로 하자
# 설치가 안된 경우에만 - npm install -g @remix-project/remixed
remixd -s . --remix-ide https://remix.ethereum.org/

# npm i ganache
npx ganache --chain.chainId 1337 --chain.networkId 1337
# 이 명령어를 실행하면 로컬 블록체인이 1337 chainId와 1337 networkId를 가지고 실행됩니다. 
# 이를 통해 개발자들은 로컬 환경에서 이더리움 스마트 계약을 테스트하고 개발할 수 있습니다.
```
